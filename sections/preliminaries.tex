\begin{thm}{(Shioda-Tate formula, \cite{ref:naskrecki2013} Theorem 3.4)}
    \label{thm:shioda}
    \begin{equation}
        \rho (S) = 2 + \sum_{v \in R} (m_{v} - 1) + \rank(E(K))
    \end{equation}
    である．
\end{thm}

\begin{thm}{(\cite{ref:naskrecki2013} Proposition 4.1.)}
    \begin{equation}
        \rank E(k(C)) + \rank E^{(u)}(k(C)) = \rank E(k(C)(\sqrt{u}))
    \end{equation}
\end{thm}
