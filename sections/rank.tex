\begin{thm}
    \begin{equation}
        \rank E_{0,s}^{(s(1 + 3s))}(\overline{\mathbb{Q}}(s)) = 1
    \end{equation}
\end{thm}
\begin{proof}
    \begin{equation}
        (s - 1, \sqrt{-1}(s - 1)) \in E_{0,s}^{(s(1 + 3s))}(\overline{\mathbb{Q}}(s))
    \end{equation}
    より rank は正である．
    \begin{equation}
        \Delta(E_{0,s}^{(s(1 + 3s))}) = 256s^{8}(s - 1)^{4}(s + 1)^{4}(3s + 1)^{6}
    \end{equation}

    上と同様に
    \begin{equation}
        \rank E_{0,s}^{(s(1 + 3s))}(\overline{\mathbb{Q}}(s)) \leq 1
    \end{equation}
\end{proof}
